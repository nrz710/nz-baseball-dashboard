function doGet(e) { var params = (e && e.parameter) || {}; var mode = params.mode || 'full'; var callback = params.callback; var cache = CacheService.getScriptCache(); // ═══ BUILD CACHE KEY from all params ═══ var cacheKey = 'dl_' + mode + '_fy=' + (params.fy || '') + '_pos=' + (params.pos || '') + '_page=' + (params.page || '1') + '_per=' + (params.per || '500'); // ═══ TRY CACHE FIRST ═══ var cached = cache.get(cacheKey); if (cached) { var result = '{"cached":true,"data":' + cached + '}'; if (callback) { return ContentService .createTextOutput(callback + '(' + result + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(result) .setMimeType(ContentService.MimeType.JSON); } // ═══ COLUMN SETS ═══ var CORE_COLS = [ 'Player','PY','SigAge','FY Age','pPos','SigTm','SigDate','FY', 'Code','ArbSys','Term','T+O','PlyrOpt?','Def?', 'GUAR','GUAR AAV','AAV','Pot IB','Pot OB', 'G','PA','HR','RBI','SB','BB%','K%','AVG','OBP','SLG', 'ISO','OPS+','wRC+','wOBA', 'xAVG','xSLG','xwOBA','Brl%','HH%','LA','SwSp%','Max EV','AvgEV', 'DRS','OAA','rWAR','fWAR','aWAR','aWAR/650' ]; // ═══ FILTERS from params ═══ var filterFY = params.fy || ''; // e.g. "2025" var filterPos = params.pos || ''; // e.g. "RF" var filterTeam= params.team|| ''; // e.g. "NYM" var minAAV = parseFloat(params.minAAV || '0'); var page = parseInt(params.page || '1'); var perPage = Math.min(parseInt(params.per || '500'), 1000); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheets = ss.getSheets(); var output = {}; sheets.forEach(function(sheet) { var name = sheet.getName(); var allData = sheet.getDataRange().getValues(); if (allData.length < 3) return; var sectionRow = allData[0]; var headers = allData[1].map(function(h) { return String(h).trim(); }); // ═══ DECIDE WHICH COLUMNS TO KEEP ═══ var keepIdx = []; if (mode === 'full') { headers.forEach(function(h, i) { keepIdx.push(i); }); } else { headers.forEach(function(h, i) { if (CORE_COLS.indexOf(h) >= 0) keepIdx.push(i); }); } // ═══ FIND COLUMN POSITIONS for filters ═══ var idxFY = headers.indexOf('FY'); var idxPos = headers.indexOf('pPos'); var idxTeam = headers.indexOf('SigTm'); var idxAAV = headers.indexOf('AAV'); // ═══ COLLECT MATCHING ROWS ═══ var matchedRows = []; for (var r = 2; r < allData.length; r++) { var row = allData[r]; var playerName = String(row[0] || '').trim(); if (!playerName || playerName === 'Player') continue; // Apply filters if (filterFY && String(row[idxFY] || '').trim() !== filterFY) continue; if (filterPos && String(row[idxPos] || '').trim() !== filterPos) continue; if (filterTeam && String(row[idxTeam]|| '').trim() !== filterTeam) continue; if (minAAV > 0) { var aavRaw = String(row[idxAAV] || '').replace(/[$,]/g, ''); var aavVal = parseFloat(aavRaw) || 0; if (aavVal < minAAV) continue; } matchedRows.push(row); } // ═══ TOTAL COUNT (for pagination UI) ═══ var totalRows = matchedRows.length; var totalPages = Math.ceil(totalRows / perPage); // ═══ PAGINATE ═══ var start = (page - 1) * perPage; var pageRows = matchedRows.slice(start, start + perPage); // ═══ BUILD SLIM OUTPUT ═══ var slim = []; slim.push(keepIdx.map(function(i) { return sectionRow[i] !== undefined ? String(sectionRow[i]) : ''; })); slim.push(keepIdx.map(function(i) { return headers[i]; })); pageRows.forEach(function(row) { slim.push(keepIdx.map(function(i) { return row[i] !== undefined ? row[i] : ''; })); }); output[name] = { rows: slim, totalRows: totalRows, totalPages: totalPages, page: page, perPage: perPage }; }); // ═══ CACHE THE RESULT (21600 = 6 hours) ═══ var json = JSON.stringify(output); try { // CacheService has a 100KB per-entry limit — only cache if small enough if (json.length < 90000) { cache.put(cacheKey, json, 21600); } } catch(err) { // Cache write failed silently — dashboard still works, just slower next time } var response = '{"cached":false,"data":' + json + '}'; if (callback) { return ContentService .createTextOutput(callback + '(' + response + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(response) .setMimeType(ContentService.MimeType.JSON); } // ═══ RUN THIS MANUALLY ONCE TO PRE-WARM THE CACHE ═══ // Also set this as a time-driven trigger: every 6 hours function warmCache() { var baseUrl = ScriptApp.getService().getUrl(); var modes = ['fast', 'full']; modes.forEach(function(mode) { var url = baseUrl + '?mode=' + mode; UrlFetchApp.fetch(url); }); }
